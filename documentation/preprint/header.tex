% Fix run-in paragraph style for ##### headers
\usepackage{titlesec}
\titleformat{\paragraph}[block]{\normalfont\normalsize\bfseries}{\theparagraph}{1em}{}
\titlespacing*{\paragraph}{0pt}{3.25ex plus 1ex minus .2ex}{0.5em}
\titleformat{\subparagraph}[block]{\normalfont\normalsize\bfseries}{\thesubparagraph}{1em}{}
\titlespacing*{\subparagraph}{0pt}{3.25ex plus 1ex minus .2ex}{0.5em}

% Table packages for tabularx with custom column types
\usepackage{booktabs}
\usepackage{tabularx}
\usepackage{array}
\newcolumntype{Y}{>{\raggedright\arraybackslash}X}

% Disable TeX hyphenation (reviewer-friendly) and avoid PDF soft-hyphen artifacts.
% Note: this does not prevent manual hyphens, only automatic word-breaking.
\usepackage{ragged2e}
\RaggedRight
\hyphenpenalty=10000
\exhyphenpenalty=10000
\pretolerance=10000
\tolerance=2000
\emergencystretch=3em

% Suppress harmless Unicode math character warnings (e.g., U+1D703 = mathematical italic theta).
% This warning occurs specifically for theta in table cells when XeLaTeX processes math.
% The character is correctly rendered via \theta in math mode using the math font;
% the warning is a false positive from XeLaTeX checking the main font for Unicode math characters.
% This appears to be specific to how Pandoc 3.x + XeLaTeX handles math in table cells.
\RequirePackage{fontspec}
\defaultfontfeatures[\rmfamily,\sffamily,\ttfamily]{}
\newfontfamily\mathfallback{TeX Gyre Termes Math}
